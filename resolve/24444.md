from collections import deque
import heapq
import sys

N,M,R = map(int,sys.stdin.readline().strip().split())

graph = {}
for n in range(1, N+1):
    graph[n] = []

for _ in range(M):
    u,v = map(int,sys.stdin.readline().strip().split())    
    heapq.heappush(graph[u],v)
    heapq.heappush(graph[v],u)

def bfs(R):
    cnt=1
    ret[R] = cnt
    queue.append(R)
    while(queue):
        q = queue.popleft()
        while True:
            if not graph[q]:
                break
            v = heapq.heappop(graph[q])
            if ret[v] == 0:
                cnt += 1
                ret[v] = cnt
                queue.append(v)

ret = [0]*(N+1)
queue = deque([])

cnt = 0
bfs(1)

for r in range(1, len(ret)):
    print(ret[r])
